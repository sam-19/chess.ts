    static forCode = function (code: number | string) {
        for (const nag of Nag.LIST) {
            if (nag.code === code
                || typeof code === 'string' && nag.code === parseInt(code)
            ) {
                return nag
            }
        }
        return null
    }
    static forSymbol = function (symbol: string) {
        for (const nag of Nag.LIST) {
            if (nag.symbol.indexOf(symbol) !== -1) {
                return nag
            }
        }
        return null
    }
